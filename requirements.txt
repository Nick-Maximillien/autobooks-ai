# Use official slim Python image
FROM python:3.10-slim

# Set working directory
WORKDIR /app

# Install system dependencies needed for wheels, OCR, and OpenCV
RUN apt-get update && apt-get install -y --no-install-recommends \
    build-essential \
    gfortran \
    libatlas3-base \
    libglib2.0-0 \
    libsm6 \
    libxext6 \
    libxrender1 \
    ffmpeg \
    tesseract-ocr \
    poppler-utils \
    && rm -rf /var/lib/apt/lists/*

# Copy requirements
COPY requirements.txt .

# Upgrade pip and install wheel-friendly packages first
RUN pip install --upgrade pip setuptools wheel \
    && pip install --no-cache-dir \
        numpy==1.26.4 \
        scipy==1.11.4 \
        scikit-image==0.25.2 \
        torch==2.1.0 \
        torchvision==0.16.0 \
        torchaudio==2.1.0 \
    && pip install --no-cache-dir -r requirements.txt

# Copy vendored libraries and weights (rarely change, cached)
COPY easyocr ./easyocr
COPY weights ./weights

# Copy the rest of the application code
COPY . .

# Ensure logs directory exists (for FileHandler)
RUN mkdir -p /app/logs

# Set PYTHONPATH so Python can find app module
ENV PYTHONPATH=/app

# Expose port
EXPOSE 8001

# Run Uvicorn pointing to app.main
CMD ["uvicorn", "app.main:app", "--host", "0.0.0.0", "--port", "8001"]
